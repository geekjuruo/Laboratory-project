\chapter{运行方式及效果}

\section{客户端运行方式}
\begin{itemize}
    \item 开发环境：macOS Mojave 10.14.5
    \item Android Studio版本：3.4
    \item 项目运行手机版本：小米 MIX 2S
\end{itemize}

\section{服务器运行方式}
\begin{itemize}
    \item 开发环境：macOS Mojave 10.14.5
    \item 运行环境：整个服务器项目在具有v4/v6双栈网络的Centos服务器上编译运行
    \item 运行方式：采用Cmake编译运行方式
\end{itemize}
